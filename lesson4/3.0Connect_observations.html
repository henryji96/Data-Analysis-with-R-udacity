Connect observations

http://ggplot2.tidyverse.org/reference/geom_path.html geom_path() connects the observations in the order in which they appear in the data. geom_line() connects them in order of the variable on the x axis. geom_step() creates a stairstep plot, highlighting exactly when changes occur. The group aesthetic determines which cases are connected together.

Example

geom_line() is suitable for time series

p1<-ggplot(economics,aes(date,unemploy))+
  geom_line()
p2<-ggplot(economics_long,
       aes(x=date,y=value01,
           color=variable))+
  geom_line()
grid.arrange(p1,p2,ncol=1)

### geom_step() is useful when you want to highlight exactly when the y value chanes

recent<-filter(economics,date>as.Date("2013-01-01"))
p1<-ggplot(recent,
       aes(date,unemploy))+
  geom_line()
p2<-ggplot(recent,
           aes(date,unemploy))+
  geom_step()
p3<-ggplot(recent,
           aes(date,unemploy))+
  geom_path(aes(color=date))
grid.arrange(p1,p2,p3,ncol=1)

# geom_path lets you explore how two variables are related over time, # e.g. unemployment and personal savings rate

m <- ggplot(economics, aes(unemploy/pop, psavert))
p1<-m + geom_point()
p2<-m + geom_path(aes(color=as.numeric(date)))
grid.arrange(p1,p2,ncol=1)

## Control line join parameters

#NAs break the line. Use na.rm = T to suppress the warning message

df

ggplot(df,aes(x,y))+
  geom_path(size=1, linejoin = "mitre",
            lineend='round',linetype=3,
            na.rm = TRUE)+
  geom_point(size=2,aes(color=x),
             na.rm=TRUE)
